  ```python
  def make_one(a):
    a = int(a)
    if a == 2 or a == 3 or a == 1:
        m[a] = 1
    elif m[a]==0:
        temp = []
        if a%3 == 0:
            temp.append(1+make_one(a/3))
        if a%2 == 0:
            temp.append(1+make_one(a/2))
        temp.append(1+make_one(a-1))
        m[a] = min(temp)
    return m[a]

n = int(input()) # 입력받은 값
m = [0]*(n+1)
  ```

  ```python
        count[num] = num
  ```
